2️⃣ Backend Python
app.py
from fastapi import FastAPI
from pydantic import BaseModel
import psycopg2
from pwdlib import PasswordHash

app = FastAPI()

password_hash = PasswordHash.recommended()

class UserRegister(BaseModel):
    nome: str
    email: str
    senha: str

class UserLogin(BaseModel):
    email: str
    senha: str


def get_conn():
    return psycopg2.connect(
        host="postgres_app",
        database="appdb",
        user="appuser",
        password="appsenha",
        port=5432
    )


@app.post("/register")
def register(user: UserRegister):

    conn = get_conn()
    cur = conn.cursor()

    hash_senha = password_hash.hash(user.senha)

    cur.execute(
        "INSERT INTO usuarios (nome,email,senha) VALUES (%s,%s,%s)",
        (user.nome, user.email, hash_senha)
    )

    conn.commit()

    cur.close()
    conn.close()

    return {"status": "usuario criado"}


@app.post("/login")
def login(user: UserLogin):

    conn = get_conn()
    cur = conn.cursor()

    cur.execute(
        "SELECT senha FROM usuarios WHERE email=%s",
        (user.email,)
    )

    result = cur.fetchone()

    if not result:
        return {"status": "usuario nao encontrado"}

    senha_hash = result[0]

    if password_hash.verify(user.senha, senha_hash):
        return {"status": "login sucesso"}

    return {"status": "senha incorreta"}

WORKDIR /app

6️⃣ Stack Docker Swarm

Arquivo stack-backend.yml

version: "3.8"

services:

  api_auth:

    image: auth_api:latest

    build: .

    networks:
      - dockermaster

    deploy:
      replicas: 2

      labels:
        - traefik.enable=true
        - traefik.http.routers.api.rule=Host(`api.inetz.com.br`)
        - traefik.http.routers.api.entrypoints=websecure
        - traefik.http.routers.api.tls=true
        - traefik.http.services.api.loadbalancer.server.port=8000

      placement:
        constraints:
          - node.role == manager

networks:
  dockermaster:
    external: true
